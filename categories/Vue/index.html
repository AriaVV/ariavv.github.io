Vue中的动画

Jelon 发表于 2017-04-05    标签：

最简单的动画

<style>
    .box{width: 100px;height: 100px;background-color:red;}
    /*动画的时间*/
    .fade-transition{transition:all 2s ease;}
    /*进入*/
    .fade-enter{opacity:0;}
    /*离开*/
    .fade-leave{opacity:0;transform:translateX(200px);}
</style>
<body>
    <script>
    window.onload = function() {
        var app = new Vue({
            data: {
                start: false
            },
            methods:{
                toggle(){
                    this.start = !this.start;    //控制.box的显示隐藏
                }
            }
        }).$mount('.container');

    };
    </script>
    <div class="container">
        <input type="button" value="按钮" @click="toggle">
        <div class="box" v-show="start" transition="fade"></div>       //transition="fade"定义动画
    </div>
</body>

利用animate.css定义动画

代码:

<script src="./libs/vue.js"></script>
<link rel="stylesheet" href="./libs/animate.css">
<title>vue动画</title>
</head>
<style>
    .container{width: 100px;margin:100px auto;}
    .box{width: 100px;height: 100px;background-color:red;}
</style>
<body>
    <script>
    window.onload = function() {
        var app = new Vue({
            data: {
                start: false
            },
            methods:{
                toggle(){
                    this.start = !this.start;
                }
            },
            // 定义所有的动画
            transitions:{
                fadeCartoon:{
                    enterClass:'zoomInLeft',  //进入,这个就是animate.css中定义的那些类名
                    leaveClass:'zoomOutRight'  //离开    
                }
            }
        }).$mount('.container');

    };
    </script>
    <div class="container">
        <input type="button" value="按钮" @click="toggle">
        <div v-show="start" class="box animated" transition="fadeCartoon"></div>
    </div>
</body>

get方式进行交互(开启服务器)

<body>    
    <script>
        window.onload = function(){
            var app = new Vue({
              el: '#box',
              data:{
                  msg:'hello world!....'
              },
              methods:{
                  getData:function(){
                      this.$http.get('get.php',{a:1,b:2}).then(function(res){
                        //成功
                          console.log(res.data);
                      },function(res){
                        失败
                        console.log(res.data);
                      })
                  }
              }
            })
        }
    </script>

    <div id="box">
        <input type="button" value="点击按钮" @click="getData()" />
    </div>
</body>

post方式进行交互(开启服务器)

post方式请求数据必须加 emulateJSON:true

    <body>
      <script>
        window.onload=function(){
            new Vue({
                el:'body',
                data:{

                },
                methods:{
                    getData:function(){
                        this.$http.post('post.php',{
                            a:1,
                            b:2
                        },{
                            emulateJSON:true
                        }).then(function(res){
                            console.log(res.data);
                        },function(res){
                            console.log(res.data);
                        });
                    }
                }
            });
        };
    </script>
    <input type="button" value="按钮" @click="getData()">
</body>

jsonp(跨域)

需要定义callback的指向 jsonp:’cb’ //定义callback的名称,默认是 callback

<body>    
    <script>
        window.onload = function() {
            var app = new Vue({
                el: '#box',
                data: {
                    msg: 'hello world!....'
                },
                methods: {
                    getData: function() {
                        //请求地址:"https://sp0.baidu.com/5a1Fazu8AA54nxGko9WTAnF6hhy/su?wd=a&cb=abc"
                        this.$http.jsonp('https://sp0.baidu.com/5a1Fazu8AA54nxGko9WTAnF6hhy/su', {
                            wd:'a'
                        }, {
                            jsonp:'cb'     //定义callback的名称
                        }).then(function(res) {
                            console.log(res.data.s);
                        }, function(res) {
                            console.log(res.data);
                        });
                    }
                }
            })
        }
    </script>

    <div id="box">
        <input type="button" value="点击按钮" @click="getData()" />

Vue中常用的简单过滤器

Jelon 发表于 2017-03-31    标签：

过滤器用来过滤模板中的数据

常用简单过滤器

语法：{{msg|filterA|filterB}}
传参语法：{{msg|filterA '参数'}}   这个参数就是你想要的参数

1、|uppercase 小写转大写

<body>    
    <script>
        window.onload = function(){
            var app = new Vue({
              el: '#box',
              data: {
                msg:'hello world'
              }
            })
        }
    </script>

    <div id="box">
        {{ msg | uppercase}}
    </div>
</body>

效果图

2、|lowercase 大写转小写

3、|capitalize 首字母大写

4、|currency 美元

5、|currency ‘￥’ 传参，人民币

6、|debounce 2000 延迟执行 一般配合事件使用

<input type="text" @keyup="show() | debounce 2000" />

7、|limitBy 2 取2条数据,跟数据配合来使用

<li v-for="val in arr | linitBy 2"></li>

8、|limitBy 2 1 从索引为1的开始，取2条数据,跟数据配合来使用

9、|filterBy ‘a’ 获取包含’a’的数据

<li v-for="val in arr | filterBy 'a' "></li>

10、|orderBy ‘a’ 排序

<li v-for="val in arr | orderBy 'a' "></li>

11、|orderBy ‘1’ 正序排列

<li v-for="val in arr | orderBy '1' "></li>

12、|orderBy ‘-1’ 倒叙排列

<li v-for="val in arr | orderBy '-1' "></li>

13、自定义过滤器

Vue.filter(名称,function(){})

例子:如果数据小于10,就在前面加上0(自定义一个名称叫做:digit的过滤器)

<body>
    <script>
        window.onload = function() {

            Vue.filter('digit',function(num){
                return num<10?'0'+num:''+num;
            })
            var app = new Vue({
                data: {
                    msg: '7'
                }
            }).$mount('.container');

        };
    </script>
    <div class="container">
        {{msg | digit}}
    </div>
</body>

14、自定义过滤器(时间戳)

<body>
    <script>
        window.onload = function() {
        Vue.filter('time',
            function(date) {
                var date = new Date(date);
                Y = date.getFullYear(),
                    m = date.getMonth() + 1,
                    d = date.getDate(),
                    H = date.getHours(),
                    i = date.getMinutes(),
                    s = date.getSeconds();
                if (m < 10) {
                    m = '0' + m;
                }
                if (d < 10) {
                    d = '0' + d;
                }
                if (H < 10) {
                    H = '0' + H;
                }
                if (i < 10) {
                    i = '0' + i;
                }
                if (s < 10) {
                    s = '0' + s;
                }
                var t = Y + '-' + m + '-' + d +  + H + ':' + i + ':' + s;
                return t;
            });

        var app = new Vue({
            data: {
                msg: Date.now()
            }
        }).$mount('.container');

        };
    </script>
    <div class="container">
        {{msg | time}}
    </div>
</body>

1、数组语法 (value, index) in arr

2、对象语法 (value, key, index) in object

value是循环的每一项的值

index是每一项的索引
<body>

        </ul>
    </div>
</body>

在对象语法中可以直接使用{{ $key }}表示键,使用{{ $index }}表示索引 `<li v-for="item in obj">{{ $key }}-{{ $index }}</li>`

自定义指令(属性)

<body>
    <script>
    // 自定义指令
    Vue.directive('red',function(color){
        this.el.style.background=color;
    });
    window.onload = function() {
        var app = new Vue({
            data: {
                msg: 'welcome',
                a:'red'
            }
        }).$mount('.container');

        };
    </script>
    <div class="container">
        <span v-red="a">11111111</span>
    </div>
</body>

自定义键盘信息

<body>
    <script>
    // 自定义绑定ctrl键,ctrl的键盘码是17
    Vue.directive('on').keyCodes.ctrl=17;
    window.onload = function() {
        var app = new Vue({
            data: {
                msg: 'welcome'
            },
            methods:{
                show:function(){
                    alert(1)
                }
            }
        }).$mount('.container');

    };
    </script>
    <div class="container">
        <input type="text" @keyup.ctrl="show()">
    </div>
</body>
