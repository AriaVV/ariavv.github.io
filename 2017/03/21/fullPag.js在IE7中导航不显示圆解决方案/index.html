修改fullpage的css样式

在设置导航器在IE7下的透明度为0
#fp-nav{+filter:alpha(opacity=0);}

固定自己设置的圆点的位置,要和原来的导航器的位置重叠
因为IE7下,是不支持点击锚点跳转的,所以只能利用导航器,这样你在点击的时候还可以跳转到下一屏
并且自己写的这个原点现在隐藏掉,如果是在IE7浏览器的情况下才显示
.circle{display:none;position: absolute;top: 50%;right:39px;margin-top: -76px; +display:block;}
.circle li{margin: 15px 0;}

JS文件

因为我是设置了每屏的导航器颜色不一样,使用JS控制的,所以在这里控制JS优化一下,点击查看
点击查自定义导航器的颜色

$(function(){
    if( $('#fullpage').length>0){
        //浏览器是IE7
        if(navigator.appName == "Microsoft Internet Explorer" && navigator.appVersion .split(";")[1].replace(/[ ]/g,"")=="MSIE6.0" || navigator.appName == "Microsoft Internet Explorer" && navigator.appVersion .split(";")[1].replace(/[ ]/g,"")=="MSIE7.0" || navigator.appName == "Microsoft Internet Explorer" && navigator.appVersion .split(";")[1].replace(/[ ]/g,"")=="MSIE8.0" || navigator.appName == "Microsoft Internet Explorer" && navigator.appVersion .split(";")[1].replace(/[ ]/g,"")=="MSIE9.0")
        {$('#fullpage').fullpage({
                resize: true,
                verticalCentered: false,
                scrollOverflow: false,
                menu:true,
                navigation: true,
                anchors: ['page1', 'page2', 'page3', 'page4'],
                afterLoad: function (anchorLink, index) {
                    $('.section').removeClass('current');
                    setTimeout(function () {
                        $('.section').eq(index -1).addClass('current');
                    }, 0);

                },
                afterRender: function(){
                    $('#part1').addClass('current')
                }
            });

        }
        //浏览器是不IE7的配置
        else {$('#fullpage').fullpage({
                resize: true,
                verticalCentered: false,
                scrollOverflow: false,
                menu:true,
                navigation: true,
                anchors: ['page1', 'page2', 'page3', 'page4'],
                afterLoad: function (anchorLink, index) {
                    $('.section').removeClass('current');
                    setTimeout(function () {
                        $('.section').eq(index -1).addClass('current');
                    }, 0);
                    if(index == 1 || index == 3){
                        $('#fp-nav span,.fp-slidesNavspan').css('background','#87c2f8');
                        $('#fp-nav li .active span,.fp-slidesNav.active span').css('background','#fff');
                        $('#fp-nav li a').on('mouseenter',function(){
                            $(this).children().css('background','#fff');
                        })
                        $('#fp-nav li a').on('mouseleave',function(){
                            $(this).children().css('background','#87c2f8');
                            $('#fp-nav li .active span').css('background','#fff');
                        })
                        $('#fp-nav span,.fp-slidesNavspan').css('+background','url(../images/y1.jpg)');
                        $('#fp-nav li .active span,.fp-slidesNav.active span').css('+background','url(../images/y2.jpg)');

                    }else if(index == 2 || index ==4){
                        $('#fp-nav span,.fp-slidesNavspan').css('background','#bfdffc');
                        $('#fp-nav li .active span,.fp-slidesNav.active span').css('background','#47a2f5');
                        $('#fp-nav li a').on('mouseenter',function(){
                            $(this).children().css('background','#47a2f5');
                        })
                        $('#fp-nav li a').on('mouseleave',function(){
                            $(this).children().css('background','#bfdffc');
                            $('#fp-nav li .active span').css('background','#47a2f5');
                        })
                    }
                },
                afterRender: function(){
                    $('#part1').addClass('current')
                }
            });
        }
})
