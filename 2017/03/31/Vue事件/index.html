Vue事件

Vue

事件

v-on:click=”函数” 基础事件
v-on可以简写成 @
例如:v-on:click="函数" ,可以写成@click="函数"

<body>
    <script>
        window.onload = function(){

            var app = new Vue({
              el: '#box',
              data: {        //data里面放着数据
                msg: true
              },
              methods: {    //方法定义在这里
                  click:function(){
                      alert(1)
                  }
              }
            })

        }
    </script>

    <div id="box"> 
        <input type="button" value="点击" v-on:click="click()">
    </div>
</body>

事件对象 $event

<body>
    <script>
        window.onload = function(){

            var app = new Vue({
              el: '#box',
              data: {
                msg: 'Hello Vue!'
              },
              methods:{
                //在html结构中点击事件传入 $event才可以使用事件对象,不然是没有效果的
                  show:function(event){
                      console.log(event)
                  }
              }
            })

        }
    </script>

    <button @click="show($event)" id="box">    
        点击按钮
    </button>
</body>

阻止事件冒泡

1、event.cancelBubble = true

<body>
    <script>
        window.onload = function(){

            var app = new Vue({
              el: '#box',
              data: {
                msg: 'Hello Vue!'
              },
              methods:{
                  show:function(event){
                      alert(1)
                      event.cancelBubble = true    //阻止冒泡
                  },
                  show2:function(event){
                      alert(2)
                  }
              }
            })

        }
    </script>

    <div @click="show2()"  id="box">
        <input type="button"  @click="show($event)"  value="点击按钮"/>
    </div>
</body>

2、@click.stop=”show()”

<body>
    <script>
        window.onload = function(){

            var app = new Vue({
              el: '#box',
              data: {
                msg: 'Hello Vue!'
              },
              methods:{
                  show:function(){
                      alert(1)
                  },
                  show2:function(){
                      alert(2)
                  }
              }
            })

        }
    </script>

    <div @click="show2()"  id="box">
        <input type="button"  @click.stop="show()"  value="点击按钮"/>     //这里阻止冒泡
    </div>
</body>

阻止默认行为

点击鼠标右键,弹出弹出框,并且鼠标右键的默认原有的功能给他禁用掉
1、 e.preventDefault();

<body>
    <script>
        window.onload = function(){

            var app = new Vue({
              el: '#box',
              data: {
                msg: 'Hello Vue!'
              },
              methods:{
                  show:function(e){
                      alert(1)
                      e.preventDefault();
                  }
              }
            })

        }
    </script>

    <input type="button"  @contextmenu="show($event)"  value="点击按钮" id="box"/>
</body>

2、 @contextmenu.prevent=”show()”

<body>
    <script>
        window.onload = function(){

            var app = new Vue({
              el: '#box',
              data: {
                msg: 'Hello Vue!'
              },
              methods:{
                  show:function(){
                      alert(1)
                  }
              }
            })

        }
    </script>

    <input type="button"  @contextmenu.prevent="show()"  value="点击按钮" id="box"/>
</body>

keyCode的使用

在Vue中,我们可以直接使用键盘码来控制行为

1、@keydown.13=”show()”
回车的键盘码是什么，按下回车键执行函数，这样我们就可以不用使用事件对象了

<body>    
    <script>
        window.onload = function(){

            var app = new Vue({
              el: '#box',
              data: {
                msg: 'Hello Vue!'
              },
              methods:{
                  show:function(){
                      alert(1)
                  }
              }
            })

        }
    </script>

    <input type="text"  @keydown.13="show()" id="box"/>
</body>

2、@keydown.enter=”show()”
按下回车键执行函数

<body>    
    <script>
        window.onload = function(){
            var app = new Vue({
              el: '#box',
              data: {
                msg: 'Hello Vue!'
              },
              methods:{
                  show:function(){
                      alert(1)
                  }
              }
            })
        }
    </script>

    <input type="text"  @keydown.enter ="show()" id="box"/>
</body>

@keydown.up=”show()” 按上键执行函数
@keydown.down=”show()” 按下键执行函数
@keydown.left=”show()” 按左键执行函数
@keydown.right=”show()” 按右键执行函数 等等,还有很多

绑定属性

在绑定属性的时候最好用v-bind
写法:<img v-bind:src="url" alt="">
简写:<img :src="url" alt="">

除了class和style所有的属性都可以 :src=”” 这种形式绑定
class在使用时候,可以:class=”[red,blue]” 这里的red是data中定义的数据;
class还可以:class=”{red:true,blue:false}”,这种指的不是data的数据,直接指向类名;
class还可以:class=”json”,json可以在data中定义
style和class的使用方式基本一致

<body>    
    <script>
        window.onload = function(){
            var app = new Vue({
              el: '#box',
              data: {
                url: 'https://ss0.bdstatic.com/5aV1bjqh_Q23odCf/static/superman/img/logo/bd_logo1_31bdc765.png'
              }
            })
        }
    </script>

    <div id="box">
        <img v-bind:src="url" alt="">
    </div>
</body>
