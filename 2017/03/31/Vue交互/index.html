Vue交互

Jelon 发表于 2017-03-31

###引包
Vue本身是不支持交互,如果想要交互,需要引入库 vue-resouce

<script src="./libs/vue.js"></script>
<script src="./libs/vue-resource.js"></script>   //交互

get方式进行交互(开启服务器)

<body>    
    <script>
        window.onload = function(){
            var app = new Vue({
              el: '#box',
              data:{
                  msg:'hello world!....'
              },
              methods:{
                  getData:function(){
                      this.$http.get('get.php',{a:1,b:2}).then(function(res){
                        //成功
                          console.log(res.data);
                      },function(res){
                        失败
                        console.log(res.data);
                      })
                  }
              }
            })
        }
    </script>

    <div id="box">
        <input type="button" value="点击按钮" @click="getData()" />
    </div>
</body>

post方式进行交互(开启服务器)

post方式请求数据必须加 emulateJSON:true

    <body>
      <script>
        window.onload=function(){
            new Vue({
                el:'body',
                data:{

                },
                methods:{
                    getData:function(){
                        this.$http.post('post.php',{
                            a:1,
                            b:2
                        },{
                            emulateJSON:true
                        }).then(function(res){
                            console.log(res.data);
                        },function(res){
                            console.log(res.data);
                        });
                    }
                }
            });
        };
    </script>
    <input type="button" value="按钮" @click="getData()">
</body>

jsonp(跨域)

需要定义callback的指向 jsonp:’cb’ //定义callback的名称,默认是 callback

<body>    
    <script>
        window.onload = function() {
            var app = new Vue({
                el: '#box',
                data: {
                    msg: 'hello world!....'
                },
                methods: {
                    getData: function() {
                        //请求地址:"https://sp0.baidu.com/5a1Fazu8AA54nxGko9WTAnF6hhy/su?wd=a&cb=abc"
                        this.$http.jsonp('https://sp0.baidu.com/5a1Fazu8AA54nxGko9WTAnF6hhy/su', {
                            wd:'a'
                        }, {
                            jsonp:'cb'     //定义callback的名称
                        }).then(function(res) {
                            console.log(res.data.s);
                        }, function(res) {
                            console.log(res.data);
                        });
                    }
                }
            })
        }
    </script>

    <div id="box">
        <input type="button" value="点击按钮" @click="getData()" />
    </div>
</body>
