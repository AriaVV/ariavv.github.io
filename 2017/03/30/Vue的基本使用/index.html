vue的基本使用

Jelon 发表于 2017-03-30

Vue不兼容低版本的IE

引包

使用vue的时候,首先需要进行引包

我们可以使用npm下载 

`npm install vue`

也可以直接链接

<script src="https://unpkg.com/vue/dist/vue.js"></script>

基本使用

我们在使用的时候首先定义HTML

<div id="box"> {{ msg }} </div>   //在这里{{ }}是用来渲染数据的

在进行声明式渲染

<script>
    window.onload = function(){
        var app = new Vue({        //定义一个vue实例,变量名字可以随便起
          el: '#box',              //这里是选择器(id选择器、标签、类名等),我们使用的是id选择器
          data: {
            msg: 'Hello Vue!'，  //这里是数据（可以放String、Number、Boolean、数组、对象等等）
            arr：['苹果','桃子','菠萝','香蕉']
          }
        })
    }
</script>

vue的常见指令

指令:扩展了html标签的属性和功能

v-model 一般用在表单元素上,用于双向数据绑定
<body>
    <script>
        window.onload = function(){
            var app = new Vue({
              el: '#box',
              data: {
                msg: 'welcome vue'
              }
            })
        }
    </script>

    <div id="box">
        <input type="text" v-model="msg">
        <input type="text" v-model="msg">
    </div>
</body>
v-for 循环

1、数组语法 (value, index) in arr

value是循环的每一项的值
index是每一项的索引
<body>
    <script>
        window.onload = function(){
            var app = new Vue({
              el: '#box',
              data: {
                arr:['苹果','桃子','菠萝','香蕉']
              }
            })
        }
    </script>

    <div id="box"> 
        <ul>
            <li v-for="(item,$index) in arr">{{ item }}-{{ $index }}</li>
        </ul>
    </div>
</body>

效果图：

如果有重复数据是添加不上的,我们可以通过track-by="$index"添加:
<li v-for="(item,$index) in arr" track-by="$index">{{ item }}-{{ $index }}</li>

2、对象语法 (value, key, index) in object

value是循环的每一项的值
key是每一项的键
index是每一项的索引
<body>
    <script>
        window.onload = function(){
            var app = new Vue({
              el: '#box',
              data: {
                obj:{a:'苹果',b:'桃子',c:'菠萝',d:'香蕉'}
              }
            })
        }
    </script>

    <div id="box"> 
        <ul>
            <li v-for="(item,$key,$index) in obj">{{ item }}-{{ $key }}-{{ $index }}</li>
        </ul>
    </div>
</body>

在对象语法中可以直接使用{{ $key }}表示键,使用{{ $index }}表示索引 `<li v-for="item in obj">{{ $key }}-{{ $index }}</li>`

效果图：

v-if 条件
<script>
    window.onload = function(){
        var app = new Vue({
          el: '#box',
          data: {
            msg: true
          }
        })
    }
</script>

<div id="box"> 
    <div v-if="msg">如果是true,就显示</div>
</div>
v-show=”true/false” 显示隐藏切换

防止闪烁

v-text=””

v-html=””

v-cloak 然后在style中添加 [v-clock]{display:none;}

自定义指令(属性)

<body>
    <script>
    // 自定义指令
    Vue.directive('red',function(color){
        this.el.style.background=color;
    });
    window.onload = function() {
        var app = new Vue({
            data: {
                msg: 'welcome',
                a:'red'
            }
        }).$mount('.container');

        };
    </script>
    <div class="container">
        <span v-red="a">11111111</span>
    </div>
</body>

效果图:

自定义元素指令(不推荐使用)

<body>
    <script>
    // 自定义元素指令
    Vue.elementDirective('span-red',{
        bind:function(){
            this.el.style.backgroundColor='red';
        }
    });
    window.onload = function() {
        var app = new Vue({
            data: {
                msg: 'welcome',
                a:'red'
            }
        }).$mount('.container');

        };
    </script>
    <div class="container">
        <span-red></span-red>
    </div>
</body>

自定义键盘信息

<body>
    <script>
    // 自定义绑定ctrl键,ctrl的键盘码是17
    Vue.directive('on').keyCodes.ctrl=17;
    window.onload = function() {
        var app = new Vue({
            data: {
                msg: 'welcome'
            },
            methods:{
                show:function(){
                    alert(1)
                }
            }
        }).$mount('.container');

    };
    </script>
    <div class="container">
        <input type="text" @keyup.ctrl="show()">
    </div>
</body>
